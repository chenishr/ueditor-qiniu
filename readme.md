####	使用配置

根据自己七牛空间的信息，在文件 php/config.php 文件中，设置以下配置项

	/* 七牛配置 */
	"qiniu-accesskey":"七牛 ak",
	"qiniu-secretkey":"七牛 sk",
	"qiniu-bucket":"七牛空间名",
	"qiniu-host":"七牛空间域名",
	/* 七牛配置 */

    "imageUrlPrefix": "七牛空间域名", /* 图片访问路径前缀 */

    "videoUrlPrefix": "七牛空间域名", /* 视频访问路径前缀 */

    "fileUrlPrefix": "七牛空间域名", /* 文件访问路径前缀 */
